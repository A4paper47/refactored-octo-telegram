# --- Admin whitelist (multiple accounts) ---
async def cmd_admin_add(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not _is_owner(update):
        return await update.message.reply_text("❌ Owner only")
    if not context.args:
        return await update.message.reply_text("Usage: /admin_add <tg_user_id> [display_name]")
    uid_raw = context.args[0].strip()
    name = " ".join(context.args[1:]).strip() or None
    if not uid_raw.isdigit():
        return await update.message.reply_text("❌ tg_user_id must be number")
    uid = int(uid_raw)
    row = AdminTelegramUser.query.filter_by(tg_user_id=uid).first()
    if not row:
        row = AdminTelegramUser(tg_user_id=uid, display_name=name, role="ADMIN", active=True)
        db.session.add(row)
    else:
        row.active = True
        if name:
            row.display_name = name
    db.session.commit()
    return await update.message.reply_text(f"✅ Admin added: {uid}")


async def cmd_admin_remove(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not _is_owner(update):
        return await update.message.reply_text("❌ Owner only")
    if not context.args:
        return await update.message.reply_text("Usage: /admin_remove <tg_user_id>")
    uid_raw = context.args[0].strip()
    if not uid_raw.isdigit():
        return await update.message.reply_text("❌ tg_user_id must be number")
    uid = int(uid_raw)
    row = AdminTelegramUser.query.filter_by(tg_user_id=uid).first()
    if not row:
        return await update.message.reply_text("⚠️ Not found")
    row.active = False
    db.session.commit()
    return await update.message.reply_text(f"✅ Admin removed: {uid}")


async def cmd_admin_list(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not _is_owner(update):
        return await update.message.reply_text("❌ Owner only")
    rows = AdminTelegramUser.query.order_by(AdminTelegramUser.active.desc(), AdminTelegramUser.created_at.desc()).all()
    if not rows:
        return await update.message.reply_text("(empty)")
    lines = ["*Admin whitelist:*", "(Owner is always admin)"]
    for r in rows:
        status = "✅" if r.active else "⛔"
        name = f" - {r.display_name}" if r.display_name else ""
        lines.append(f"{status} `{r.tg_user_id}`{name}")
    return await update.message.reply_text("\n".join(lines), parse_mode=ParseMode.MARKDOWN)
